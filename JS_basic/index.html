# js中的预解析

  1. 变量申明提前了

      console.log(num) //输出 undefined, 变量的申明提前到了当前所在作用域的最上面，变量申明未赋值就是undefined
  var num = 10

  2. 函数的申明也会提前

      f1() //输出hello

  function f1() {   // 申明提前到当前所在作用域的最上面
    console.log('hello')
  }

  3. 例子1

     f1()
 var num = 20

 function f1() {
   console.log(num) // 输出undefined, 是函数作用域内的num变量的申明提前了
   var num = 10
 }

    例子2

    
 f1()//函数调用的时候会把函数声明提前到作用域上面
 var num = 20//声明会被提前到使用之前

 function f1() {
   console.log(num) //  还是输出undefined 仔细体会函数与num的预解析
   // var num = 10
 }

    上面代码预解析后

     var num
 function f1() {
   console.log(num) //  输出undefined
   // var num = 10
 }

 f1()
 num = 20

  4. 例子

      function f1() {
      console.log(num)//undefined
      var num = 10
  }
  f1()
  console.log(num)//报错
  • javascript解析器执行javascript代码的时候分为两个过程预解析过程和执行过程
    • 预解析过程：
      • 把变量的声明提升到当前作用域的最前面，只提升申明不提升赋值
      • 把函数的声明提升到当前作用域的最前面，只提升声明不提升调用
      • 先提升变量再提升function

  1. 例子

        var a = 25

    function abc() {
      alert(a) //undefined
      var a = 10
    }
    abc()
    console.log(a) // 25

  2. 例子

    
  console.log(a) //函数体

  function a() {
    console.log('aaa')
  }
  var a = 1
  console.log(a) //1

    上面代码预解析后

    
  var a

  function a() {
    console.log('aaa')
  }

  console.log(a) //函数体,命名函数 重新给a赋值了
  a = 1
  console.log(a) //1

    注意下面的例子

    
  console.log(a) //undefined

  var m = function a() {
    console.log('aaa')
  }
  console.log(a) //undefined
  console.log(m) //函数体
  m() // 输出 aaa
  var a = 1
  console.log(a) //1

    上面代码预解析后

      
  var a
  var m
  console.log(a) //undefined

  m = function a() {
    console.log('aaa')
  }

  console.log(a) //undefined
  console.log(m) //函数体
  m() //aaa
  a = 1
  console.log(a) //1

  4. 例子

    f1()
console.log(c) //9
console.log(b) //9
console.log(a) //报错

function f1() {
  var a = b = c = 9
  console.log(a) //9
  console.log(b) //9
  console.log(c) //9
}

    上面代码解析后

    
  function f1() {
    var a
    a = 9 //局部变量
    b = 9 //隐式全局
    c = 9 //隐式全局

    console.log(a) //9
    console.log(b) //9
    console.log(c) //9
  }
  f1()
  console.log(c) //9
  console.log(b) //9
  console.log(a) //报错

  5. 例子

  }

  var f1
