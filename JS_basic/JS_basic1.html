# ES6 学习

# let

  1. let 定义的变量有块级作用域
  var foo = "bar"

  if (true) {
    var foo = 'baz'
  }

  console.log(foo) //var 定义的变量没有块级作用域，所以这里输出baz。 与python一样没有块级作用域

  console.log('-----------------')

  let foo1 = "bar"

  if (true) {
    let foo1 = 'baz'
  }

  console.log(foo1) //let 定义的变量是有块级作用域的，所以这里输出bar
  1. let 定义的变量没有变量提升

console.log(foo) //var定义的变量有定义的提升，所以输出undefined

var foo = 'bar'


console.log(foo) //let定义的变量没有定义的提升，直接报错了

let foo = 'bar'
  1. 在区块中,使用let命令声明变量之前，该变量都是不可用的。let声明的变量会绑定当前块级区域，只能先声明再使用。
  var tmp = 123;
  if (true) {
    tmp = "abc";  //下面let声明了一个局部变量(有块级作用域)，在这里未声明就使用，报错。let声明变量没有提升一说
    let tmp;     
  }
  1. 在同一个块级作用域中，不允许重复声明同一个变量，以下代码错误
  function func1() {
    let a = 10;
    let a = 2
  }

  function func2() {
    let a = 10;
    var a = 2
  }

  

### const

- 用来声明一个常量，声明的同时必须初始化，一旦声明以后不能修改，和let一样有块级作用域，不存在变量提升，不允许重复声明
- 能用const的情况下，少用let，var就不要使用了
